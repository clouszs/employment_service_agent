### 3.4 database.py - 数据库连接

#### 作用
管理MySQL数据库连接，提供Session管理，支持异步操作。

#### 实现步骤

**步骤1：定义数据库引擎**

```python
# 文件：backend/app/core/database.py

from sqlalchemy import create_engine
from sqlalchemy.orm import sessionmaker, Session
from sqlalchemy.ext.declarative import declarative_base
from contextlib import contextmanager
from typing import Generator

from app.core.config import get_settings

settings = get_settings()

# 数据库连接URL
DATABASE_URL = f"mysql+pymysql://{settings.MYSQL_USER}:{settings.MYSQL_PASSWORD}@{settings.MYSQL_HOST}:{settings.MYSQL_PORT}/{settings.MYSQL_DATABASE}?charset=utf8mb4"

# 创建引擎
engine = create_engine(
    DATABASE_URL,
    pool_size=10,
    max_overflow=20,
    pool_pre_ping=True,  # 连接池健康检查
    echo=settings.DEBUG  # SQL日志
)

# Session工厂
SessionLocal = sessionmaker(autocommit=False, autoflush=False, bind=engine)

# 模型基类
Base = declarative_base()


def get_db() -> Generator[Session, None, None]:
    """
    获取数据库Session（依赖注入）
    
    使用方式：
    @router.get("/users")
    def get_users(db: Session = Depends(get_db)):
        ...
    """
    db = SessionLocal()
    try:
        yield db
    finally:
        db.close()


@contextmanager
def get_db_context():
    """上下文管理器方式获取Session"""
    db = SessionLocal()
    try:
        yield db
        db.commit()
    except Exception:
        db.rollback()
        raise
    finally:
        db.close()


def init_db():
    """初始化数据库（创建所有表）"""
    from app.models import Base
    Base.metadata.create_all(bind=engine)


def drop_db():
    """删除所有表（仅用于开发测试）"""
    Base.metadata.drop_all(bind=engine)
```

**步骤2：添加Redis连接**

```python
# 添加到 database.py

import redis
from typing import Optional

_redis_client: Optional[redis.Redis] = None


def get_redis() -> redis.Redis:
    """获取Redis客户端"""
    global _redis_client
    if _redis_client is None:
        _redis_client = redis.Redis(
            host=settings.REDIS_HOST,
            port=settings.REDIS_PORT,
            password=settings.REDIS_PASSWORD,
            decode_responses=True,
            socket_timeout=5
        )
    return _redis_client


def close_redis():
    """关闭Redis连接"""
    global _redis_client
    if _redis_client:
        _redis_client.close()
        _redis_client = None
```

```python
def get_current_user(
    credentials: HTTPAuthorizationCredentials = Depends(security),
    db: Session = Depends(get_db)
) -> SysUser:
    """
    获取当前登录用户
    
    使用方式：
    @router.get("/me")
    def get_me(current_user: SysUser = Depends(get_current_user)):
        return current_user
    """
    token = credentials.credentials
    
    # 验证Token
    payload = verify_token(token)
    if payload is None:
        raise HTTPException(
            status_code=status.HTTP_401_UNAUTHORIZED,
            detail="无效的认证令牌"
        )
    
    user_id = payload.get("sub")
    if user_id is None:
        raise HTTPException(
            status_code=status.HTTP_401_UNAUTHORIZED,
            detail="令牌格式错误"
        )
    
    # 查询用户
    user = db.query(SysUser).filter(SysUser.id == int(user_id)).first()
    if user is None:
        raise HTTPException(
            status_code=status.HTTP_401_UNAUTHORIZED,
            detail="用户不存在"
        )
    
    if user.status != 1:
        raise HTTPException(
            status_code=status.HTTP_403_FORBIDDEN,
            detail="用户已被禁用"
        )
    
    return user


def get_current_user_optional(
    credentials: Optional[HTTPAuthorizationCredentials] = Depends(security),
    db: Session = Depends(get_db)
) -> Optional[SysUser]:
    """可选的用户认证（允许未登录访问）"""
    if credentials is None:
        return None
    
    try:
        return get_current_user(credentials, db)
    except HTTPException:
        return None


def get_admin_user(
    current_user: SysUser = Depends(get_current_user)

    """获取管理员用户（需要管理员权限）"""
```

```python
# 文件：backend/app/core/embedding.py

import httpx
from typing import List, Optional
import numpy as np
from functools import lru_cache

from app.core.config import get_settings
from app.core.database import get_redis

settings = get_settings()


class EmbeddingClient:
    """
    Embedding客户端
    
    实现要点：
    1. 支持单个和批量生成
    2. Redis缓存Embedding
    3. 自动重试机制
    """
    
    API_URL = "https://dashscope.aliyuncs.com/api/v1/services/embeddings/text-embedding/text-embedding"
    
    def __init__(self):
        self.api_key = settings.DASHSCOPE_API_KEY
        self.model = settings.EMBEDDING_MODEL
        self.dimension = settings.EMBEDDING_DIMENSION
    
    def get_embedding(self, text: str) -> List[float]:
        """获取单个文本的Embedding"""
        
        # 先检查缓存
        redis_client = get_redis()
        cache_key = f"emb:{hash(text)}"
        cached = redis_client.get(cache_key)
        
        if cached:
            return json.loads(cached)
        
        # 调用API
        headers = {
            "Authorization": f"Bearer {self.api_key}",
            "Content-Type": "application/json"
        }
        
        payload = {
            "model": self.model,
            "input": {"texts": [text]},
            "parameters": {"text_type": "query"}
        }
        
        response = httpx.post(
            self.API_URL,
            headers=headers,
            json=payload,
            timeout=30.0
        )
        response.raise_for_status()
        
        result = response.json()
        embedding = result["output"]["embeddings"][0]["embedding"]
        
        # 缓存结果（24小时）
        redis_client.setex(cache_key, 86400, json.dumps(embedding))
        
        return embedding
    
    def get_embeddings_batch(self, texts: List[str]) -> List[List[float]]:
        """批量获取Embedding"""
        
        headers = {
            "Authorization": f"Bearer {self.api_key}",
            "Content-Type": "application/json"
        }
        
        payload = {
            "model": self.model,
            "input": {"texts": texts},
            "parameters": {"text_type": "document"}
        }
        
        response = httpx.post(
            self.API_URL,
            headers=headers,
            json=payload,
            timeout=60.0
        )
        response.raise_for_status()
        
        result = response.json()
        embeddings = [e["embedding"] for e in result["output"]["embeddings"]]
        
        return embeddings
    
    def cosine_similarity(
        self,
        vec1: List[float],
        vec2: List[float]
    ) -> float:
        """计算Cosine相似度"""
        vec1 = np.array(vec1)
        vec2 = np.array(vec2)
        
        dot_product = np.dot(vec1, vec2)
        norm1 = np.linalg.norm(vec1)
        norm2 = np.linalg.norm(vec2)
        
        return dot_product / (norm1 * norm2 + 1e-8)
```

### 4.6 semantic_chunker.py - 语义切分

#### 作用
基于Embedding相似度进行智能切分，确保每个chunk语义完整。

#### 实现步骤

```python
# 文件：backend/app/core/semantic_chunker.py

from typing import List, Dict, Any
import numpy as np

from app.core.embedding import embedding_client
from app.core.config import get_settings

settings = get_settings()


class SemanticChunker:
    """
    语义感知的文档切分器
    
    核心思想：
    1. 先按句子切分（保持句子完整性）
    2. 计算相邻句子的语义相似度
    3. 在相似度"断点"处进行切分
    4. 确保每个chunk语义完整
    
    面试要点：
    - 展示对传统切分方式局限性的理解
    - 展示对语义完整性的重视
    """
    
    def __init__(
        self,
        similarity_threshold: float = 0.7,
        min_chunk_size: int = 100,
        max_chunk_size: int = 500
    ):
        self.similarity_threshold = similarity_threshold
        self.min_chunk_size = min_chunk_size
        self.max_chunk_size = max_chunk_size
    
    def chunk(self, text: str, metadata: Dict[str, Any] = None) -> List[Dict[str, Any]]:
        """
        语义切分
        
        Args:
            text: 待切分文本
            metadata: 元数据（会附加到每个chunk）
        
        Returns:
            chunk列表，每个包含 text, metadata
        """
        # 1. 按句子切分
        sentences = self._split_sentences(text)
        
        if len(sentences) <= 1:
            return [{"text": text, "metadata": metadata or {}}]
        
        # 2. 计算句子Embedding
        embeddings = embedding_client.get_embeddings_batch(sentences)
        
        # 3. 计算相邻句子相似度
        similarities = self._calculate_adjacent_similarity(embeddings)
        
        # 4. 找到语义断点
        breakpoints = self._find_breakpoints(similarities)
        
        # 5. 根据断点创建chunks
        chunks = self._create_chunks(sentences, breakpoints, metadata)
        
        return chunks
    
    def _split_sentences(self, text: str) -> List[str]:
        """按句子切分"""
        separators = ["。", "！", "？", "；", "\n"]
        
        sentences = []
        current = ""
        
        for char in text:
            current += char
            if char in separators:
                if current.strip():
                    sentences.append(current.strip())
                current = ""
        
        if current.strip():
            sentences.append(current.strip())
        
        return sentences
    
    def _calculate_adjacent_similarity(
        self,
        embeddings: List[List[float]]
    ) -> List[float]:
        """计算相邻句子相似度"""
        similarities = []
        
        for i in range(len(embeddings) - 1):
            sim = embedding_client.cosine_similarity(
                embeddings[i],
                embeddings[i + 1]
            )
            similarities.append(sim)
        
        return similarities
    
    def _find_breakpoints(self, similarities: List[float]) -> List[int]:
        """找到语义断点（相似度低于阈值的位置）"""
        breakpoints = []
        
        for i, sim in enumerate(similarities):
            if sim < self.similarity_threshold:
                breakpoints.append(i + 1)
        
        return breakpoints
    
    def _create_chunks(
        self,
        sentences: List[str],
        breakpoints: List[int],
        metadata: Dict[str, Any]
    ) -> List[Dict[str, Any]]:
        """根据断点创建chunks"""
        chunks = []
        
        boundaries = [0] + breakpoints + [len(sentences)]
        
        for i in range(len(boundaries) - 1):
            start = boundaries[i]
            end = boundaries[i + 1]
            
            chunk_sentences = sentences[start:end]
            chunk_text = "".join(chunk_sentences)
            
            # 检查chunk大小
            if len(chunk_text) < self.min_chunk_size:
                continue
            elif len(chunk_text) > self.max_chunk_size:
                # 需要进一步切分
                sub_chunks = self._split_large_chunk(chunk_text, metadata)
                chunks.extend(sub_chunks)
            else:
                chunk_metadata = {
                    **(metadata or {}),
                    "chunk_type": "semantic",
                    "sentence_count": len(chunk_sentences)
                }
                chunks.append({
                    "text": chunk_text,
                    "metadata": chunk_metadata
                })
        
        return chunks
    
    def _split_large_chunk(
        self,
        text: str,
        metadata: Dict[str, Any]
    ) -> List[Dict[str, Any]]:
        """切分过大的chunk"""
        chunks = []
        sentences = self._split_sentences(text)
        
        current = ""
        for sentence in sentences:
            if len(current) + len(sentence) <= self.max_chunk_size:
                current += sentence
            else:
                if current:
                    chunks.append({
                        "text": current,
                        "metadata": {
                            **(metadata or {}),
                            "chunk_type": "split"
                        }
                    })
                current = sentence
        
        if current:
            chunks.append({
                "text": current,
                "metadata": {
                    **(metadata or {}),
                    "chunk_type": "split"
                }
            })
        
        return chunks


# 全局语义切分器
semantic_chunker = SemanticChunker()
```

```python
        生成节点：带引用的回答生成

        1. 构建Prompt（包含检索结果）

        return {
            "response": response.content,

        response = state["response"]

        issues = self.fact_verifier.verify(response)
        
        return {
            "fact_issues": issues
        }
    
    @traceable(name="agent.refuse", tags=["agent", "refusal"])
```

```python
from langgraph.graph import StateGraph, END, START
from langgraph.checkpoint.memory import MemorySaver
from typing import Literal

from app.agent.state import AgentState
from app.agent.nodes import AgentNodes


class AgentGraph:
    """
    LangGraph工作流构建
    
    工作流结构：
    START → route → [search | tool_call | direct | refuse]
    
    search路径：
    route → search → check_confidence → [generate | refuse]
    generate → check_consistency → verify_facts → END
    
    面试要点：
    - 展示对状态机设计的理解
    - 解释条件分支的实现
    - Checkpoint的作用（暂停/恢复）
    """
    
    def __init__(self):
        self.nodes = AgentNodes()
        self.graph = None
        self.checkpointer = MemorySaver()
    
    def build(self) -> StateGraph:
        """构建工作流图"""
        
        # 创建状态图
        graph = StateGraph(AgentState)
        
        # 添加节点
        graph.add_node("route", self.nodes.route_query)
        graph.add_node("search", self.nodes.search_knowledge)
        graph.add_node("check_confidence", self.nodes.check_confidence)
        graph.add_node("generate", self.nodes.generate_response)
        graph.add_node("check_consistency", self.nodes.check_consistency)
        graph.add_node("verify_facts", self.nodes.verify_facts)
        graph.add_node("refuse", self.nodes.generate_refusal)
        
        # 添加边（定义流程）
        
        # START → route
        graph.add_edge(START, "route")
        
        # route → 分支
        graph.add_conditional_edges(
            "route",
            self._route_decision,
            {
                "search": "search",
                "tool_call": "tool_call",  # 暂未实现
                "direct": END,
                "refuse": "refuse"
            }
        )
        
        # search → check_confidence
        graph.add_edge("search", "check_confidence")
        
        # check_confidence → 分支
        graph.add_conditional_edges(
            "check_confidence",
            self._confidence_decision,
            {
                "accept": "generate",
                "refuse": "refuse"
            }
        )
        
        # generate → check_consistency
        graph.add_edge("generate", "check_consistency")
        
        # check_consistency → verify_facts
        graph.add_edge("check_consistency", "verify_facts")
        
        # verify_facts → END
        graph.add_edge("verify_facts", END)
        
        # refuse → END
        graph.add_edge("refuse", END)
        
        self.graph = graph
        return graph
    
    def _route_decision(self, state: AgentState) -> Literal["search", "tool_call", "direct", "refuse"]:
        """路由决策函数"""
        return state["route"]
    
    def _confidence_decision(self, state: AgentState) -> Literal["accept", "refuse"]:
        """置信度决策函数"""
        if state["should_refuse"]:
            return "refuse"
        return "accept"
    
    def compile(self):
        """编译工作流"""
        graph = self.build()
        return graph.compile(checkpointer=self.checkpointer)
    
    async def run(self, query: str, conversation_id: int = None, user_id: int = None) -> AgentState:
        """
        运行Agent
        
        Args:
            query: 用户查询
            conversation_id: 会话ID
            user_id: 用户ID
        
        Returns:
            最终状态
        """
        app = self.compile()
        
        # 初始状态
        initial_state = AgentState(
            messages=[],
            current_query=query,
            search_results=[],
            citations=[],
            confidence=0.0,
            query_risk_level="medium",
            tools_used=[],
            tool_outputs={},
            response="",
            route="",
            should_refuse=False,
            refusal_reason="",
            consistency_issues=[],
            fact_issues=[],
            version_info={},
            temporal_warnings=[],
            conversation_id=conversation_id or 0,
            user_id=user_id or 0,
            created_at=datetime.now()
        )
        
        # 执行工作流
        result = await app.ainvoke(initial_state)
        
        return result
```

### 5.6 hallucination_defense.py - 幻觉防御系统

#### 作用
实现五重防护机制：动态阈值、置信度过滤、自我一致性、事实核验。

#### 实现步骤

```python
# 文件：backend/app/agent/hallucination_defense.py

from typing import Dict, Any, List, Tuple, Optional
from enum import Enum
import re
from datetime import datetime

from app.core.llm import llm_client


class QueryRiskLevel(Enum):
    """查询风险等级"""
    HIGH = "high"
    MEDIUM = "medium"
    LOW = "low"


class DynamicConfidenceThreshold:
    """
    动态置信度阈值
    
    面试要点：
    - 展示对业务场景的理解
    - 不同查询类型用不同阈值
    """
    
    THRESHOLD_CONFIG = {
        QueryRiskLevel.HIGH: {
            "min_confidence": 0.80,
            "min_results": 3,
            "require_citation": True
        },
        QueryRiskLevel.MEDIUM: {
            "min_confidence": 0.65,
            "min_results": 2,
            "require_citation": True
        },
        QueryRiskLevel.LOW: {
            "min_confidence": 0.40,
            "min_results": 1,
            "require_citation": False
        }
    }
    
    def classify_query(self, query: str) -> QueryRiskLevel:
        """分类查询风险等级"""
        high_keywords = ["落户", "补贴", "政策", "规定", "流程"]
        low_keywords = ["你好", "谢谢", "再见"]
        
        query_lower = query.lower()
        
        if any(k in query_lower for k in high_keywords):
            return QueryRiskLevel.HIGH
        if any(k in query_lower for k in low_keywords):
            return QueryRiskLevel.LOW
        return QueryRiskLevel.MEDIUM
    
    def should_accept_result(
        self,
        query: str,
        confidence: float,
        results_count: int,
        has_citation: bool
    ) -> Tuple[bool, Optional[str]]:
        """判断检索结果是否可接受"""
        
        risk_level = self.classify_query(query)
        config = self.THRESHOLD_CONFIG[risk_level]
        
        reasons = []
        
        if confidence < config["min_confidence"]:
            reasons.append(f"置信度{confidence:.2f}低于阈值{config['min_confidence']}")
        
        if results_count < config["min_results"]:
            reasons.append(f"检索结果数{results_count}低于最小要求{config['min_results']}")
        
        if config["require_citation"] and not has_citation:
            reasons.append("该类查询必须有引用来源")
        
        if reasons:
            return False, "; ".join(reasons)
        
        return True, None


class SelfConsistencyChecker:
    """
    自我一致性检查器
    
    检查同一对话中同类问题回答是否一致
    """
    
    async def check(
        self,
        query: str,
        response: str,
        history: List[Dict[str, Any]]
    ) -> Tuple[bool, List[Dict[str, Any]]]:
        """检查一致性"""
        
        # 找到相似的查询
        similar_queries = self._find_similar(query, history)
        
        if not similar_queries:
            return True, []
        
        issues = []
        for sq in similar_queries:
            # 使用LLM判断是否矛盾
            contradiction = await self._detect_contradiction(
                response,
                sq["answer"],
                query
            )
            
            if contradiction:
                issues.append({
                    "current_query": query,
                    "previous_query": sq["query"],
                    "contradiction_type": contradiction["type"],
                    "severity": contradiction["severity"]
                })
```

```python
from dataclasses import dataclass

from app.core.llm import llm_client

@dataclass
class SentenceCitation:
    """句子级别引用"""
    sentence: str
    sentence_index: int
    source_chunk_id: str
    source_text: str
    support_type: str  # direct/indirect/none
    confidence: float


class SentenceLevelCitationTracker:
    """
    句子级别引用追踪器
    
    面试要点：
    - 展示对引用精度的追求
    - 每个句子都能追溯到来源
    """
    
    async def track(
        self,
        response: str,
        search_results: List[Dict[str, Any]]
    ) -> List[Dict[str, Any]]:
        """
        追踪引用
        
        流程：
        1. 切分回答为句子
        2. 为每个句子找到最相关来源
        3. 验证支持度
        """
        sentences = self._split_sentences(response)
        
        citations = []
        for i, sentence in enumerate(sentences):
            # 找到最相关来源
            best_match = self._find_best_source(sentence, search_results)
            
            # 验证支持度
            support_info = await self._verify_support(sentence, best_match)
            
            citation = {
                "ref_id": f"[{i+1}]",
                "sentence": sentence,
                "sentence_index": i,
                "chunk_id": best_match.get("chunk_id", ""),
                "source_text": best_match.get("text", ""),
                "support_type": support_info["type"],
                "support_confidence": support_info["confidence"]
            }
            
            citations.append(citation)
        
        return citations
    
    def _split_sentences(self, text: str) -> List[str]:
        """切分为句子"""
        separators = ["。", "！", "？", "；"]
        
        sentences = []
        current = ""
        
        for char in text:
            current += char
            if char in separators:
                if current.strip():
                    sentences.append(current.strip())
                current = ""
        
        if current.strip():
            sentences.append(current.strip())
        
        return sentences
    
    def _find_best_source(
        self,
        sentence: str,
        sources: List[Dict]
    ) -> Dict:
        """找到最相关来源"""
        sentence_emb = embedding_client.get_embedding(sentence)
        
        best_match = {}
        best_similarity = -1
        
        for source in sources:
            source_emb = embedding_client.get_embedding(source["text"])
            similarity = embedding_client.cosine_similarity(sentence_emb, source_emb)
            
            if similarity > best_similarity:
                best_similarity = similarity
                best_match = source
        
        return best_match
    
    async def _verify_support(
        self,
        sentence: str,
        source: Dict
    ) -> Dict:
        """验证支持度"""
        if not source:
            return {"type": "none", "confidence": 0}
        
        prompt = f"""
判断来源是否支持句子。

句子：{sentence}
来源：{source['text']}

输出类型：direct/indirect/none
"""
        
        result = llm_client.chat(prompt)
        
        support_type = result.content.strip().lower()
        
        confidence_map = {
            "direct": 0.9,
            "indirect": 0.6,
            "none": 0.2
        }
        
        return {
            "type": support_type,
            "confidence": confidence_map.get(support_type, 0.5)
        }
```

### 5.8 temporal_retriever.py - 时效感知检索

#### 作用
检索时考虑文档时效性，新文档权重更高，过期文档降权。

#### 实现步骤

```python
# 文件：backend/app/agent/temporal_retriever.py

import math
from datetime import datetime, timedelta
from typing import List, Dict, Any
from dataclasses import dataclass

from app.core.vectorstore import vector_store


@dataclass
class TemporalSearchResult:
    """带时效性的检索结果"""
    chunk_id: str
    text: str
    score: float
    temporal_score: float
    combined_score: float
    valid_until: datetime
    is_expired: bool
    freshness_level: str


class TemporalAwareRetriever:
    """
    时效感知检索器
    
    面试要点：
    - 展示对知识时效性的重视
    - 检索时考虑时间维度
    """
    
    CONFIG = {
        "half_life_days": 180,
        "similarity_weight": 0.7,
        "temporal_weight": 0.3,
        "expired_penalty": 0.1
    }
    
    async def retrieve(
        self,
        query: str,
        top_k: int = 10
    ) -> List[TemporalSearchResult]:
        """
        时效感知检索
        
        流程：
        1. 基础向量检索
        2. 计算时效性得分
        3. 计算综合得分
        4. 过期文档降权
        """
        # 基础检索
        base_results = vector_store.search(query, top_k=top_k * 2)
        
        today = datetime.now()
        temporal_results = []
        
        for r in base_results:
            # 计算时效性得分
            temporal_score = self._calculate_temporal_score(
                r.metadata,
                today
            )
            
            # 综合得分
            combined_score = (
                r.score * self.CONFIG["similarity_weight"] +
                temporal_score * self.CONFIG["temporal_weight"]
            )
            
            # 判断过期状态
            valid_until = r.metadata.get("valid_until")
            if valid_until:
                valid_until = datetime.fromisoformat(valid_until)
                is_expired = valid_until < today
            else:
                valid_until = None
                is_expired = False
            
            # 过期降权
            if is_expired:
                combined_score *= self.CONFIG["expired_penalty"]
            
            # 新鲜度级别
            freshness = self._classify_freshness(valid_until, is_expired)
            
            temporal_results.append(TemporalSearchResult(
                chunk_id=r.chunk_id,
                text=r.text,
                score=r.score,
                temporal_score=temporal_score,
                combined_score=combined_score,
                valid_until=valid_until,
                is_expired=is_expired,
                freshness_level=freshness
            ))
        
        # 按综合得分排序
        temporal_results.sort(key=lambda x: x.combined_score, reverse=True)
        
        return temporal_results[:top_k]
    
    def _calculate_temporal_score(
        self,
        metadata: Dict,
        today: datetime
    ) -> float:
        """计算时效性得分"""
        valid_from = metadata.get("valid_from")
        
        if valid_from:
            valid_from = datetime.fromisoformat(valid_from)
            days_since = (today - valid_from).days
        else:
            days_since = 0
        
        half_life = self.CONFIG["half_life_days"]
        
        # 指数衰减
        freshness = math.exp(-0.693 * days_since / half_life)
        
        return freshness
    
    def _classify_freshness(
        self,
        valid_until: datetime,
        is_expired: bool
    ) -> str:
        """分类新鲜度"""
        if is_expired:
            return "expired"
        
        if valid_until is None:
            return "unknown"
        
        days_until = (valid_until - datetime.now()).days
        
        if days_until > 90:
            return "fresh"
        elif days_until > 30:
            return "aging"
        else:
            return "expiring_soon"
```

```python
        self,
        file_path: str,
        title: str,
        category: str,
        source: str,
        valid_from: Optional[datetime] = None,
        valid_until: Optional[datetime] = None,
        user_id: int = None

        """
        上传文档
        
        流程：
        1. 解析文档内容
        2. 语义切分
        3. 存储到MySQL
        4. 索引到向量库

            category=category,
            source=source,

            valid_until=valid_until,

        self.db.commit()

            content,
            metadata={
                "document_id": document.id,
                "doc_title": title,
                "category": category,
                "valid_from": valid_from.isoformat() if valid_from else None,
                "valid_until": valid_until.isoformat() if valid_until else None
            }
        )

        for i, chunk in enumerate(chunks):
            chunk_record = KbChunk(
                document_id=document.id,
                content=chunk["text"],
                chunk_index=i,
                char_count=len(chunk["text"])
            )

        self.db.add_all(chunk_records)
        self.db.commit()
        
        # 6. 索引到向量库
        chunk_ids = vector_store.add_documents(chunks)
        
        # 7. 更新chunk的vector_id
        for i, chunk_record in enumerate(chunk_records):
            chunk_record.vector_id = chunk_ids[i]
        
        self.db.commit()
        
        # 8. 更新文档chunk_count
        document.chunk_count = len(chunks)
        self.db.commit()
        
        return document

            ).delete()
            
            # 标记文档为删除
            document.status = 0
            self.db.commit()
    
    def _calculate_hash(self, file_path: str) -> str:

        with open(file_path, "rb") as f:
            for chunk in f.iter_content(8192):
                sha256.update(chunk)
        
        return sha256.hexdigest()
```

```python
        if not verify_password(password, user.password_hash):
            raise ValueError("密码错误")
        
        if user.status != 1:
            raise ValueError("用户已被禁用")
        
        # 生成Token
        token = create_access_token(user.id, user.role)
        
        # 更新登录信息
        user.last_login_at = datetime.now()
        self.db.commit()
        
        return {
            "user_id": user.id,
            "username": user.username,
            "real_name": user.real_name,
            "role": user.role,
            "token": token
        }
    
    def get_user(self, user_id: int) -> Optional[SysUser]:
        """获取用户"""
        return self.db.query(SysUser).filter(
            SysUser.id == user_id
        ).first()
    
    def update_user(
        self,
        user_id: int,
        real_name: Optional[str] = None,
        email: Optional[str] = None,
        phone: Optional[str] = None
    ) -> SysUser:
        """更新用户信息"""
        
        user = self.get_user(user_id)
        
        if real_name:
            user.real_name = real_name
        if email:
            user.email = email
        if phone:
            user.phone = phone
        
        self.db.commit()
        self.db.refresh(user)
        
        return user
    
    def change_password(
        self,
        user_id: int,
        old_password: str,
        new_password: str
    ):
        """修改密码"""
        
        user = self.get_user(user_id)
        
        if not verify_password(old_password, user.password_hash):
            raise ValueError("原密码错误")
        
        user.password_hash = hash_password(new_password)
        self.db.commit()
```

### 10.3 health_monitor.py - 知识库健康监控

#### 作用
每日检查知识库健康度，发送过期告警。

#### 实现步骤

```python
# 文件：backend/app/monitor/health_monitor.py

from sqlalchemy.orm import Session
from datetime import datetime, timedelta
from typing import List, Dict, Any
import math

from app.models.knowledge import KbDocument
from app.models.monitor import KbHealthLog


class HealthMonitor:
    """
    知识库健康监控
    
    面试要点：
    - 展示可观测性思维
    - 自动化运维能力
    """
    
    CONFIG = {
        "warning_days": 30,
        "freshness_half_life": 180,
        "min_health_score": 60
    }
    
    def __init__(self, db: Session):
        self.db = db
    
    async def run_daily_check(self) -> Dict[str, Any]:
        """每日健康检查"""
        
        today = datetime.now()
        
        # 检查即将过期文档
        warning_docs = self._check_upcoming_expiry(today)
        
        # 检查已过期文档
        expired_docs = self._check_expired_docs(today)
        
        # 计算健康度
        health_score = self._calculate_health_score(today)
        
        # 生成报告
        report = {
            "check_date": today.strftime("%Y-%m-%d"),
            "health_score": health_score,
            "warning_count": len(warning_docs),
            "expired_count": len(expired_docs),
            "warning_docs": warning_docs,
            "expired_docs": expired_docs
        }
        
        # 记录日志
        log = KbHealthLog(
            check_date=today,
            total_docs=self._get_total_docs(),
            current_docs=self._get_current_docs(),
            warning_docs=len(warning_docs),
            expired_docs=len(expired_docs),
            health_score=health_score
        )
        
        self.db.add(log)
        self.db.commit()
        
        return report
    
    def _check_upcoming_expiry(self, today: datetime) -> List[Dict]:
        """检查即将过期文档"""
        warning_date = today + timedelta(days=self.CONFIG["warning_days"])
        
        docs = self.db.query(KbDocument).filter(
            KbDocument.valid_until.between(today, warning_date),
            KbDocument.is_current == 1,
            KbDocument.status == 1
        ).all()
        
        return [
            {
                "id": d.id,
                "title": d.title,
                "valid_until": d.valid_until.strftime("%Y-%m-%d"),
                "days_until_expiry": (d.valid_until - today).days
            }
            for d in docs
        ]
    
    def _check_expired_docs(self, today: datetime) -> List[Dict]:
        """检查已过期文档"""
        docs = self.db.query(KbDocument).filter(
            KbDocument.valid_until < today,
            KbDocument.is_current == 1,
            KbDocument.status == 1
        ).all()
        
        return [
            {
                "id": d.id,
                "title": d.title,
                "days_expired": (today - d.valid_until).days
            }
            for d in docs
        ]
    
    def _calculate_health_score(self, today: datetime) -> float:
        """计算健康度"""
        docs = self.db.query(KbDocument).filter(
            KbDocument.is_current == 1,
            KbDocument.status == 1
        ).all()
        
        if not docs:
            return 100.0
        
        total_score = 0
        
        for d in docs:
            if d.valid_from:
                days_since = (today - d.valid_from).days
                freshness = math.exp(
                    -0.693 * days_since / self.CONFIG["freshness_half_life"]
                )
            else:
                freshness = 0.5
            
            if d.valid_until and d.valid_until < today:
                freshness *= 0.1
            
            total_score += freshness
        
        health_score = (total_score / len(docs)) * 100
        
        return round(health_score, 2)
```
