```python
from passlib.context import CryptContext

pwd_context = CryptContext(
    schemes=["bcrypt", "argon2"],
    deprecated="auto",
    bcrypt__rounds=12,
    argon2__time_cost=3,
    argon2__memory_cost=65536
)

def hash_password(plain: str) -> str:
    """生成密码哈希"""
    return pwd_context.hash(plain)

def verify_password(plain: str, stored: Optional[str]) -> bool:
    """校验密码"""
    if not stored:
        return False
    try:
        return pwd_context.verify(plain, stored)
    except Exception:
        return False
```

**优化**:
```python
from functools import lru_cache
from typing import Optional

# 用户信息缓存（生产环境建议使用Redis）
_user_cache: dict[int, SysUser] = {}

def get_current_user(
    credentials: HTTPAuthorizationCredentials = Depends(bearer_scheme),
    db: Session = Depends(get_db),
) -> SysUser:
    """从 Bearer Token 解析并返回当前用户。"""
    token = credentials.credentials
    try:
        payload = decode_access_token(token)
        user_id = int(payload.get("sub"))
        
        # 检查缓存（生产环境建议设置TTL）
        if user_id in _user_cache:
            return _user_cache[user_id]
        
        user = db.query(SysUser).filter(
            SysUser.id == user_id,
            SysUser.status == 1
        ).first()
        
        if user is None:
            raise HTTPException(status_code=status.HTTP_401_UNAUTHORIZED, detail="用户不存在或已禁用")
        
        # 更新缓存
        _user_cache[user_id] = user
        return user
    except (jwt.PyJWTError, TypeError, ValueError):
        raise HTTPException(status_code=status.HTTP_401_UNAUTHORIZED, detail="无效或过期的令牌")
```

```python
from functools import lru_cache
from typing import Optional
import threading

# 带最大缓存大小的客户端管理
class OpenAIClientManager:
    _instance: Optional['OpenAIClientManager'] = None
    _lock = threading.Lock()
    
    def __init__(self):
        self._embedding_client: Optional[OpenAI] = None
        self._llm_client: Optional[OpenAI] = None
    
    @classmethod
    def get_instance(cls) -> 'OpenAIClientManager':
        if cls._instance is None:
            with cls._lock:
                if cls._instance is None:
                    cls._instance = cls()
        return cls._instance
    
    def get_embedding_client(self) -> OpenAI:
        if self._embedding_client is None:
            self._embedding_client = OpenAI(
                api_key=settings.dashscope_api_key, 
                base_url=settings.dashscope_base_url
            )
        return self._embedding_client
    
    def get_llm_client(self) -> OpenAI:
        if self._llm_client is None:
            self._llm_client = OpenAI(
                api_key=settings.dashscope_api_key, 
                base_url=settings.dashscope_base_url
            )
        return self._llm_client

_client = OpenAIClientManager.get_instance
```

```python
from openai import RateLimitError, APITimeoutError, APIError

@retry(

def _create_embedding_with_retry(client: OpenAI, model: str, input_texts: list[str]) -> Any:
    """带重试的嵌入创建"""
    try:
        return client.embeddings.create(
            model=model,
            input=input_texts,
            dimensions=settings.embedding_dim,
            encoding_format="float",
        )
    except Exception as e:
        logger.error(f"Embedding API error: {str(e)}")
        raise

def embed_texts(texts: list[str]) -> list[list[float]]:
    """批量向量化，返回与输入顺序一致的向量列表"""
    if not texts:
        return []
    
    out: list[list[float]] = []
    for i in range(0, len(texts), _MAX_BATCH):
        batch = texts[i : i + _MAX_BATCH]
        resp = _create_embedding_with_retry(_client(), settings.embedding_model, batch)
        
        items = sorted(resp.data, key=lambda d: d.index)
        out.extend(item.embedding for item in items)
    
    return out
```

```python
import re

# 敏感词缓存（生产环境建议使用Redis）
_sensitive_words_cache: list[str] = []
_cache_timestamp: float = 0
_CACHE_TTL: int = 300  # 5分钟缓存

# 编译的正则表达式缓存
_word_patterns: re.Pattern = None

def _get_sensitive_words(db: Session) -> list[str]:
    """获取敏感词列表（带缓存）"""
    global _sensitive_words_cache, _cache_timestamp, _word_patterns
    
    now = time.time()
    if not _sensitive_words_cache or (now - _cache_timestamp) > _CACHE_TTL:
        words = db.execute(
            select(OpSensitiveWord.word).where(OpSensitiveWord.status == 1)
        ).scalars().all()
        _sensitive_words_cache = [w for w in words if w]
        _cache_timestamp = now
        
        # 预编译正则表达式
        if _sensitive_words_cache:
            pattern = "|".join(re.escape(w) for w in _sensitive_words_cache)
            _word_patterns = re.compile(pattern)
    
    return _sensitive_words_cache

def moderate(db: Session, text: str) -> dict:
    """敏感词前置检查"""
    words = _get_sensitive_words(db)
    
    # 使用正则匹配
    if _word_patterns:
        matches = _word_patterns.findall(text)
        hits = list(set(matches))  # 去重
    else:
        hits = [w for w in words if w in text]
    
    if not hits:
        return {"action": 0, "masked": text, "hits": []}
    
    # 替换敏感词
    masked = text
    for w in hits:
        masked = masked.replace(w, "***")
    
    # 获取最严处理动作
    actions = []
    for w in words:
        if w in hits:
            word_obj = db.execute(
                select(OpSensitiveWord).where(OpSensitiveWord.word == w)
            ).scalar_one_or_none()
            if word_obj:
                actions.append(word_obj.action)
    
    action = 1 if 1 in actions else (2 if 2 in actions else 3)
    return {"action": action, "masked": masked, "hits": hits}
```
